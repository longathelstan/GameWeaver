' --- STANDARD FUNCTIONS (Do not change) ---
' These functions are required for the quiz to run in PowerPoint.
' They handle slide navigation and showing feedback.

Sub ShowQuestion(slideIndex As Integer)
    ' This function navigates to the specified question slide.
    If slideIndex > 0 And slideIndex <= ActivePresentation.Slides.Count Then
        ActivePresentation.SlideShowWindow.View.GotoSlide slideIndex
    Else
        MsgBox "Error: Invalid slide index."
    End If
End Sub

Sub ShowFeedback(isCorrect As Boolean, correctAnswer As String)
    ' This function shows a message box with feedback.
    If isCorrect Then
        MsgBox "Correct! Well done."
    Else
        MsgBox "Sorry, that's not correct. The right answer was: " & correctAnswer
    End If
End Sub
